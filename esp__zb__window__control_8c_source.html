72void esp_zb_app_signal_handler(esp_zb_app_signal_t *signal_struct)
73{
74 // Указатель на код сигнала (uint32_t)
75 uint32_t *p_sg_p = signal_struct->p_app_signal;
76 // Статус выполнения последней операции
77 esp_err_t err_status = signal_struct->esp_err_status;
78 // Тип сигнала (получаем разыменованием указателя)
79 esp_zb_app_signal_type_t sig_type = *p_sg_p;
80 switch (sig_type) {
81 // Сигнал, генерируемый ZDO (Zigbee Device Object) при пропуске нормального запуска.
82 // Указывает, что нужно самостоятельно инициализировать стек.
83 case ESP_ZB_ZDO_SIGNAL_SKIP_STARTUP:
84 // Запускаем начальную инициализацию (считывание сетевых параметров из NVS и т.п.)
85 ESP_LOGI(TAG, "Initialize Zigbee stack");
86 esp_zb_bdb_start_top_level_commissioning(ESP_ZB_BDB_MODE_INITIALIZATION);
87 break;
88 // Сигнал первого запуска устройства (чистый flash) или перезагрузки после сохранения сети.
89 case ESP_ZB_BDB_SIGNAL_DEVICE_FIRST_START:
90 case ESP_ZB_BDB_SIGNAL_DEVICE_REBOOT:
91 if (err_status == ESP_OK) {
92 // Инициализируем драйвер лампы
93 ESP_LOGI(TAG, "Deferred driver initialization %s", deferred_driver_init() ? "failed" : "successful");
94 ESP_LOGI(TAG, "Device started up in %s factory-reset mode", esp_zb_bdb_is_factory_new() ? "" : "non");
95 // Проверяем, является ли устройство "заводски новым" (нет сохранённой сети)
96 if (esp_zb_bdb_is_factory_new()) {
97 ESP_LOGI(TAG, "Start network steering");
98 // Запускаем поиск и присоединение к существующей Zigbee-сети (Network Steering)
99 esp_zb_bdb_start_top_level_commissioning(ESP_ZB_BDB_MODE_NETWORK_STEERING);
100 } else {
101 ESP_LOGI(TAG, "Device rebooted");
102 // Устройство уже было в сети, просто перезагрузилось; никаких дополнительных действий не требуется.
103 }
104 } else {
105 // Ошибка на этапе инициализации Zigbee стека
106 /* commissioning failed */
107 ESP_LOGW(TAG, "Failed to initialize Zigbee stack (status: %s)", esp_err_to_name(err_status));
108 }
109 break;
110 // Сигнал, завершающий процесс Network Steering (удачный или нет)
111 case ESP_ZB_BDB_SIGNAL_STEERING:
112 if (err_status == ESP_OK) {
113 // Успешное присоединение к сети – выводим информацию о параметрах сети
114 esp_zb_ieee_addr_t extended_pan_id;
115 esp_zb_get_extended_pan_id(extended_pan_id);
116 ESP_LOGI(TAG, "Joined network successfully (Extended PAN ID: %02x:%02x:%02x:%02x:%02x:%02x:%02x:%02x, PAN ID: 0x%04hx, Channel:%d, Short Address: 0x%04hx)",
117 extended_pan_id[7], extended_pan_id[6], extended_pan_id[5], extended_pan_id[4],
118 extended_pan_id[3], extended_pan_id[2], extended_pan_id[1], extended_pan_id[0],
119 esp_zb_get_pan_id(), esp_zb_get_current_channel(), esp_zb_get_short_address());
120 } else {
121 // Присоединение не удалось – планируем повтор через 1 секунду (1000 мс)
122 ESP_LOGI(TAG, "Network steering was not successful (status: %s)", esp_err_to_name(err_status));
123 esp_zb_scheduler_alarm((esp_zb_callback_t)bdb_start_top_level_commissioning_cb, ESP_ZB_BDB_MODE_NETWORK_STEERING, 1000);
124 }
125 break;
126 default:
127 ESP_LOGI(TAG, "ZDO signal: %s (0x%x), status: %s", esp_zb_zdo_signal_to_string(sig_type), sig_type,
128 esp_err_to_name(err_status));
129 break;
130 }
131}

140static esp_err_t zb_attribute_handler(const esp_zb_zcl_window_covering_movement_message_t *message)
141{
142 esp_err_t ret = ESP_OK;
143 ESP_RETURN_ON_FALSE(message, ESP_FAIL, TAG, "Empty message");
144 ESP_RETURN_ON_FALSE(message->info.status == ESP_ZB_ZCL_STATUS_SUCCESS, ESP_ERR_INVALID_ARG, TAG, "Error status(%d)", message->info.status);
145
146 ESP_LOGI(TAG, "Received: cluster 0x%x, command 0x%x, data size %d",
147 message->info.cluster, message->command, message->payload.lift_value);
148
149 if (message->info.dst_endpoint == HA_ESP_WINDOW_COVEWRING_ENDPOINT) {
150 if (message->info.cluster == ESP_ZB_ZCL_CLUSTER_ID_WINDOW_COVERING) {
151 switch (message->command) {
152 case 0x02: // открыть
153 ESP_LOGI(TAG, "Open command: both servos open sequentially");
154 servo_driver_open_both_sequential();
155 break;
156
157 case 0x01: // закрыть
158 ESP_LOGI(TAG, "Close command: both servos close sequentially");
159 servo_driver_close_both_sequential();
160 break;
161
162 case 0x05: // открыть на процент
163 ESP_LOGI(TAG, "Move to %d%%: Servo1 full open, Servo2 to percent", message->payload.lift_value);
164 servo_driver_move_with_percentage(message->payload.lift_value);
165 break;
166
167 default:
168 ESP_LOGW(TAG, "Unknown command: 0x%x", message->command);
169 break;
170 }
171 }
172 }
173 return ret;
174}

182static esp_err_t zb_action_handler(esp_zb_core_action_callback_id_t callback_id, const void *message)
183{
184 esp_err_t ret = ESP_OK;
185 switch (callback_id) {
186 case ESP_ZB_CORE_SET_ATTR_VALUE_CB_ID:
187 // Пришла команда на изменение атрибута – передаём в специализированный обработчик
188 ret = zb_attribute_handler((esp_zb_zcl_window_covering_movement_message_t *)message);
189 break;
190 case ESP_ZB_CORE_WINDOW_COVERING_MOVEMENT_CB_ID: // 0x30 - получена команда
191 // Обработка принятых команд (включая команды для window covering)
192 ret = zb_attribute_handler((esp_zb_zcl_window_covering_movement_message_t *)message);
193 break;
194 default:
195 // Другие типы callback (например, чтение атрибута, команда) – просто логируем
196 ESP_LOGW(TAG, "Receive Zigbee action(0x%x) callback", callback_id);
197 break;
198 }
199 return ret;
200}

206static void esp_zb_task(void *pvParameters)
207{
208 /* initialize Zigbee stack */
209 /* 1. Инициализация стека Zigbee с конфигурацией End Device (ZED) */
210 esp_zb_cfg_t zb_nwk_cfg = ESP_ZB_ZED_CONFIG();
211 esp_zb_init(&zb_nwk_cfg);
212 /* 2. Создание конечной точки (endpoint)
213 * Макрос ESP_ZB_DEFAULT_WINDOW_COVERING_CONFIG предоставляет стандартные настройки:
214 * - Включает кластеры: Basic, Identify, Groups, Scenes, On/off.
215 */
216
217 esp_zb_window_covering_cfg_t window_covering_cfg = ESP_ZB_DEFAULT_WINDOW_COVERING_CONFIG() ;
218 window_covering_cfg.window_cfg.covering_type = 0x00;
219 esp_zb_ep_list_t *esp_zb_window_covering_ep = esp_zb_window_covering_ep_create(HA_ESP_WINDOW_COVEWRING_ENDPOINT , &window_covering_cfg);
220 /* 3. Добавление пользовательской информации о производителе и модели в кластер Basic */
221 zcl_basic_manufacturer_info_t info = {
222 .manufacturer_name = ESP_MANUFACTURER_NAME,
223 .model_identifier = ESP_MODEL_IDENTIFIER,
224 };
225
226 esp_zcl_utility_add_ep_basic_manufacturer_info(esp_zb_window_covering_ep, HA_ESP_WINDOW_COVEWRING_ENDPOINT, &info);
227
228 /* 4. Регистрация созданного endpoint'а в Zigbee стеке */
229 esp_zb_device_register(esp_zb_window_covering_ep);
230
231 /* 5. Регистрация единого обработчика всех действий (callback) */
232 esp_zb_core_action_handler_register(zb_action_handler);
233
234 /* 6. Установка маски первичных каналов (например, все каналы) */
235 esp_zb_set_primary_network_channel_set(ESP_ZB_PRIMARY_CHANNEL_MASK);
236
237 /* 7. Запуск Zigbee стека.
238 * Параметр false означает, что BDB commissioning не стартует автоматически,
239 * а будет запущен позже через сигналы (ESP_ZB_ZDO_SIGNAL_SKIP_STARTUP).
240 */
241 ESP_ERROR_CHECK(esp_zb_start(false));
242
243 /* 8. Вход в бесконечный цикл обработки событий стека (никогда не возвращает управление) */
244 esp_zb_stack_main_loop();
245}
