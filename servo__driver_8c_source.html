61void servo_driver_set_angle(uint8_t servo_id, float angle)
62{
63 if (!s_is_initialized) {
64 ESP_LOGW(TAG, "Servo not initialized");
65 return;
66 }
67 if (servo_id >= s_channel_count) {
68 ESP_LOGE(TAG, "Invalid servo_id %d", servo_id);
69 return;
70 }
71 if (angle < 0) angle = 0;
72 if (angle > SERVO_MAX_ANGLE) angle = SERVO_MAX_ANGLE;
73
74 uint8_t channel = (servo_id == 0) ? s_servo1_ch : s_servo2_ch;
75 esp_err_t ret = iot_servo_write_angle(s_speed_mode, channel, angle);
76 if (ret != ESP_OK) {
77 ESP_LOGE(TAG, "Failed to set servo%d angle %.1f°", servo_id+1, angle);
78 } else {
79 ESP_LOGD(TAG, "Servo%d angle set to %.1f°", servo_id+1, angle);
